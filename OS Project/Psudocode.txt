// Define constants
CONSTANT SIZE = 15
CONSTANT CELL_NUMBER = 52
CONSTANT MAX_PLAYER = 4
CONSTANT MAX_DE = 6

// Define classes and structures
CLASS Cell
    ATTRIBUTES:
        INTEGER x, y, value

CLASS Token
    ATTRIBUTES:
        INTEGER x, y, index, id, hitrate
        BOOLEAN revstep

CLASS Player
    ATTRIBUTES:
        INTEGER team, complete, turnsWO6
        ARRAY OF Token goti[4]
        STRING color

// Define global variables
ARRAY OF Cell cells[CELL_NUMBER]
ARRAY OF Cell houses[MAX_PLAYER][4]
ARRAY OF Player players[MAX_PLAYER]
ARRAY OF Player* Winners
ARRAY OF INTEGER turn
ARRAY OF INTEGER DiceRolls[4][3]
INTEGER numTokens, nbPlayer
BOOLEAN gamefinished
ARRAY OF INTEGER killvalues[3]

// Define functions
FUNCTION createMaster(INT turnval, INT choice)
FUNCTION Masterthread(VOID* args)
FUNCTION initGame()
FUNCTION menu(REFERENCE INT choice, REFERENCE INT players)
FUNCTION play()
FUNCTION playerT(VOID* arg)
FUNCTION displayCurrent()
FUNCTION shuff(REFERENCE ARRAY OF INT turn)

// Main function
FUNCTION main()
    // Initialize mutex
    INITIALIZE pthread_mutex_t m1

    // Initialize Winners array
    FOR i = 0 TO 3
        APPEND NULL TO Winners

    // Main game loop
    WHILE true
        INTEGER choice = -1
        INTEGER Numplayers = -1
        menu(choice, Numplayers)

        IF choice == 2 THEN
            OUTPUT "Stopping the game ..."
            BREAK
        END IF

        nbPlayer = Numplayers
        play()

        // Print winners
        FOR i = 0 TO Winners.length
            IF Winners[i] != NULL THEN
                OUTPUT i + 1 + " - " + Winners[i].color + "Player " + Winners[i].team + 1 + " won!"
            END IF
        END FOR
    END WHILE

    // Destroy mutex
    pthread_mutex_destroy(&m1)
    RETURN 0
END FUNCTION

FUNCTION createMaster(turnval = 0, choice = -1)
    DECLARE master
    DECLARE args = make_pair(turnval, choice)
    pthread_create(&master, NULL, Masterthread, (void *)&args)
    DECLARE status
    pthread_join(master, &status)
    RETURN status
END FUNCTION

FUNCTION initGame()
    DECLARE newCells[CELL_NUMBER][2]
    DECLARE newHouses[MAX_PLAYER][4][2]

    FOR i = 0 TO CELL_NUMBER - 1
        cells[i].y = newCells[i][1]
        cells[i].x = newCells[i][0]
        cells[i].value = 0
    END FOR

    DECLARE c = 0
    FOR i = 0 TO MAX_PLAYER - 1
        FOR j = 0 TO 3
            houses[i][j].y = newHouses[i][j][1]
            houses[i][j].x = newHouses[i][j][0]
            houses[i][j].value = (j + 1) + (i) * 10

            players[i].goti[j].x = houses[i][j].x
            players[i].goti[j].y = houses[i][j].y
            players[i].goti[j].index = -1
            players[i].team = i
            players[i].goti[j].id = 'a' + c
            INCREMENT c
        END FOR
        players[i].setColor()
    END FOR
END FUNCTION

FUNCTION menu(choice, players)
    WHILE choice > 2 OR choice < 1
        OUTPUT "1 - To Play"
        OUTPUT "2 - To Quit"
        OUTPUT ">"
        INPUT choice
    END WHILE

    IF choice == 1 THEN
        WHILE players < 2 OR players > 4
            OUTPUT "How many players (between 2 and 4)?"
            OUTPUT ">"
            INPUT players
        END WHILE

        numTokens = -1
        WHILE numTokens < 1 OR numTokens > 4
            OUTPUT "How many Tokens (between 2 and 4)?"
            OUTPUT ">"
            INPUT numTokens
        END WHILE
    END IF
END FUNCTION




FUNCTION play()
    initGame()
    DECLARE Playthreads[4]
    srand(time(NULL))

    FOR i = 0 TO nbPlayer - 1
        turn.push_back(i) // Player turn : 0, 1, 2, or 3
    END FOR
    shuff(turn) // Generate random order for turns

    WHILE NOT gamefinished DO
        FOR i = 0 TO nbPlayer - 1
            pthread_create(&Playthreads[i], NULL, playerT, (void *)&turn[i])
        END FOR
        FOR i = 0 TO nbPlayer - 1
            pthread_join(Playthreads[i], NULL)
        END FOR
        shuff(turn)
    END WHILE

    RETURN
END FUNCTION



FUNCTION Masterthread(args)
    INT l = args.first
    INT m = args.second

    Player p = players[l]
    token currtok = NULL
    IF m != -1 THEN
        currtok = p.goti[m]
    END IF

    INT turnval = p.team

    // Exit game for that player if 20 turns without rolling a 6
    IF p.turnsWO6 >= 20 THEN
        PRINT p.color + "\nPlayer " + (p.team + 1) + " has played 20 turns without a 6. He is out of the game."
        REMOVE(turn, turnval)
        nbPlayer--
        FOR i = SIZE_OF(Winners) - 1 DOWNTO 0
            IF Winners[i] == NULL THEN
                Winners[i] = NEW Player(p)
                BREAK
            END IF
        END FOR
        IF nbPlayer == 1 THEN
            FOR i = SIZE_OF(Winners) - 1 DOWNTO 0
                IF Winners[i] == NULL THEN
                    Winners[i] = NEW Player(players[turn[0]])
                    BREAK
                END IF
            END FOR
            gamefinished = TRUE
        END IF
    END IF

    // Check for entry square in home column
    IF currtok != NULL AND currtok.isFinalWay(turnval) AND currtok.hitrate >= 1 THEN
        currtok.index = 100
    END IF

    IF currtok != NULL THEN
        // Check if token was hit
        INT c = 0
        INT xval = -1
        INT yval = -1
        FOR i = 0 TO MAX_PLAYER - 1
            IF i != turnval THEN
                FOR j = 0 TO numTokens - 1
                    IF players[i].goti[j].x == currtok.x AND players[i].goti[j].y == currtok.y AND players[i].team != p.team THEN
                        IF NOT players[i].goti[j].isSafe() THEN
                            c++
                            xval = i
                            yval = j
                        END IF
                        IF c >= 2 THEN
                            BREAK
                        END IF
                    END IF
                END FOR
                IF c >= 2 THEN
                    BREAK
                END IF
            END IF
        END FOR

        // Winner check code
        IF p.complete == numTokens THEN
            BOOL already = FALSE
            FOR i = 0 TO SIZE_OF(Winners) - 1
                IF Winners[i] != NULL AND Winners[i].team == p.team THEN
                    already = TRUE
                    BREAK
                END IF
            END FOR
            IF NOT already THEN
                FOR i = 0 TO SIZE_OF(Winners) - 1
                    IF Winners[i] == NULL THEN
                        PRINT "Add normal " + p.color + p.team + RESET + " in winner pos : " + i
                        Winners[i] = NEW Player(p)
                        SLEEP(5)
                        BREAK
                    END IF
                END FOR
                REMOVE(turn, p.team)
                nbPlayer--
                PRINT "Erased " + p.color + p.team + 1 + RESET
                IF nbPlayer == 1 THEN
                    FOR i = SIZE_OF(Winners) - 1 DOWNTO 0
                        IF Winners[i] == NULL THEN
                            PRINT "Add Final " + players[turn[0]].color + players[turn[0]].team + RESET + " in winner pos : " + i
                            Winners[i] = NEW Player(players[turn[0]])
                            SLEEP(5)
                            BREAK
                        END IF
                    END FOR
                    gamefinished = TRUE
                END IF
            END IF
        END IF
    END IF

    RETURN NULL
END FUNCTION



FUNCTION playerT(arg)
    ACQUIRE_LOCK(m1)
    PRINT "Turn : "
    FOR i = 0 TO SIZE_OF(turn)
        PRINT turn[i] + " "
    END FOR
    PRINT NEW_LINE

    // Extract player index from argument
    INT l = CAST_TO_INT(arg)
    Player p = players[l]

    INT turnval = p.team
    INT_ARRAY MyRolls = DiceRolls[turnval]
    FOR i = 0 TO 2
        MyRolls[i] = 0 // Clear rolls array before next own move
    END FOR

    displayCurrent()

    PRINT NEW_LINE
    PRINT "It's up to player " + p.color + " " + (turnval + 1) + " to play."
    PRINT NEW_LINE

    MyRolls[0] = RANDOM(MAX_DE) + 1
    IF MyRolls[0] == 6 THEN
        p.turnsWO6 = 0
        MyRolls[1] = RANDOM(MAX_DE) + 1
        IF MyRolls[1] == 6 THEN
            MyRolls[2] = RANDOM(MAX_DE) + 1
            IF MyRolls[2] == 6 THEN
                PRINT "Launch of the dice ... Result : 6,6,6"
                PRINT "PASS TURN"
                FOR i = 0 TO 2
                    MyRolls[i] = 0
                END FOR
                createMaster(turnval)
                SLEEP(1)
                RELEASE_LOCK(m1)
                RETURN NULL
            END IF
        END IF
    ELSE
        p.turnsWO6++
    END IF

    PRINT "Launch of the dice ... Result : "
    FOR i = 0 TO 2
        IF MyRolls[i] != 0 THEN
            PRINT MyRolls[i] + ","
        END IF
    END FOR
    PRINT "\b \n"

    IF MyRolls[0] != 6 THEN
        BOOL allin = TRUE
        FOR i = 0 TO numTokens
            IF p.goti[i].index != -1 AND p.goti[i].id != 5 THEN
                allin = FALSE
                BREAK
            END IF
        END FOR
        IF allin THEN
            PRINT "PASS TURN"
            SLEEP(1)
            createMaster(turnval)
            RELEASE_LOCK(m1)
            RETURN NULL
        END IF
    END IF

    BOOL pass = FALSE
    INT choice = 0
    FOR i = 0 TO 2
        IF MyRolls[i] == 0 THEN
            CONTINUE
        END IF
        BOOL allin = TRUE
        FOR i = 0 TO numTokens
            IF p.goti[i].index != -1 AND p.goti[i].id != 5 THEN
                allin = FALSE
                BREAK
            END IF
        END FOR
        IF allin AND MyRolls[0] != 6 THEN
            PRINT "PASS TURN"
            SLEEP(1)
            createMaster(turnval)
            RELEASE_LOCK(m1)
            RETURN NULL
        END IF
        WHILE pass == FALSE
            PRINT "Which piece do you want to move a throw of " + MyRolls[i]
            PRINT NEW_LINE
            PRINT ">"
            READ choice
            IF choice < 1 OR choice > numTokens THEN
                PRINT "INVALID: Please enter a number between 1 and " + numTokens + "."
            ELSE IF p.goti[choice - 1].id == 5 THEN
                PRINT "INVALID: This piece has already home."
            ELSE IF p.goti[choice - 1].index == -1 AND MyRolls[i] != 6 THEN
                PRINT "INVALID: This piece is in the house."
            ELSE
                pass = TRUE
            END IF
        END WHILE
        pass = FALSE
        INT result = MyRolls[i]
        token curr = p.goti[choice - 1]

        createMaster(turnval, choice - 1)

        IF curr.index == -1 AND result == 6 THEN
            curr.index = p.getIndexByTurn()
            curr.x = cells[p.getIndexByTurn()].x
            curr.y = cells[p.getIndexByTurn()].y
        ELSE IF curr.index != -1 THEN
            IF curr.index + result > 106 THEN
                PRINT "Cannot move without exact number."
                PRINT "PASS TURN"
                createMaster(turnval, choice - 1)
                SLEEP(1)
                RELEASE_LOCK(m1)
                RETURN NULL
            END IF

            INT step = result
            WHILE step > 0
                createMaster(turnval, choice - 1)
                IF curr.index < CELL_NUMBER THEN
                    curr.index = (curr.index + 1) MOD CELL_NUMBER
                    curr.x = cells[curr.index].x
                    curr.y = cells[curr.index].y
                ELSE
                    curr.index++
                    p.moveOnFinalWay(choice)

                    IF curr.index == 106 THEN
                        displayCurrent()
                        PRINT "MADE IT TO THE END"
                        p.complete++
                        createMaster(turnval, choice - 1)
                        IF gamefinished THEN
                            RELEASE_LOCK(m1)
                            RETURN NULL
                        END IF
                        curr.id = 5
                        BOOL allin = TRUE
                        FOR i = 0 TO numTokens
                            IF p.goti[i].index != -1 AND p.goti[i].id != 5 THEN
                                allin = FALSE
                                BREAK
                            END IF
                        END FOR
                        IF allin THEN
                            PRINT "PASS TURN"
                            createMaster(turnval, choice - 1)
                            RELEASE_LOCK(m1)
                            RETURN NULL
                        END IF
                    END IF
                END IF
                createMaster(turnval, choice - 1)
                IF killvalues[0] != 0 THEN
                    INT c = killvalues[0]
                    INT xval = killvalues[1]
                    INT yval = killvalues[2]
                    IF c == 1 AND step == 1 THEN
                        PRINT "Token Hit"
                        PRINT p.color + "Killer Team: " + (p.team + 1) + " " + players[xval].color
                        PRINT NEW_LINE + "Victim Team: " + (players[xval].team + 1)
                        players[xval].goti[yval].index = -1
                        players[xval].goti[yval].x = houses[xval][yval].x
                        players[xval].goti[yval].y = houses[xval][yval].y
                        curr.hitrate++
                    END IF
                    IF c >= 2 THEN
                        IF curr.index < CELL_NUMBER THEN
                            curr.index = (curr.index - 1) MOD CELL_NUMBER
                            curr.x = cells[curr.index].x
                            curr.y = cells[curr.index].y
                        END IF
                    END IF
                END IF
                step--
            END WHILE
        END IF
    END FOR

    createMaster(turnval, choice - 1)
    RELEASE_LOCK(m1)
    RETURN NULL
END FUNCTION


FUNCTION displayCurrent()
    // Define a character array to represent the current board
    CHARACTER_ARRAY currentBoard[SIZE][SIZE]

    // Copy the elements of the 'Grid' array to 'currentBoard'
    FOR i = 0 TO SIZE
        FOR j = 0 TO SIZE
            currentBoard[i][j] = Grid[i][j]
        END FOR
    END FOR

    // Update 'currentBoard' with player positions
    FOR i = 0 TO MAX_PLAYER
        FOR j = 0 TO 4
            IF players[i].goti[j].id != 5 THEN
                // Retrieve current position reference from 'currentBoard'
                CHARACTER REFERENCE currposition = currentBoard[players[i].goti[j].y][players[i].goti[j].x]

                // Check if multiple players are at the same position
                IF currposition >= 'a' AND currposition <= 'z' THEN
                    currposition = 'T'
                ELSE
                    currposition = players[i].goti[j].id
                END IF
            END IF
        END FOR
    END FOR

    // Display the updated board
    PRINT NEW_LINE
    FOR i = 0 TO SIZE
        PRINT " "
        FOR j = 0 TO SIZE
            STRING s = ""
            s = s + currentBoard[i][j] + " "

            SWITCH currentBoard[i][j] DO
                CASE '1':
                    s = "O "
                    s = RED + s + RESET
                    BREAK
                CASE '2':
                    s = "O "
                    s = GREEN + s + RESET
                    BREAK
                CASE '3':
                    s = "O "
                    s = CYAN + s + RESET
                    BREAK
                CASE '4':
                    s = "O "
                    s = YELLOW + s + RESET
                    BREAK
                DEFAULT:
                    s = currentBoard[i][j]
                    s += " "
                    FOR a = 0 TO MAX_PLAYER
                        FOR b = 0 TO 4
                            IF i == players[a].goti[b].y AND j == players[a].goti[b].x THEN
                                SWITCH players[a].team DO
                                    CASE 0:
                                        s = RED + s + RESET
                                        BREAK
                                    CASE 1:
                                        s = GREEN + s + RESET
                                        BREAK
                                    CASE 2:
                                        s = CYAN + s + RESET
                                        BREAK
                                    CASE 3:
                                        s = YELLOW + s + RESET
                                        BREAK
                                END SWITCH
                            END IF
                        END FOR
                    END FOR
                    BREAK
            END SWITCH
            PRINT s
        END FOR
        PRINT NEW_LINE
    END FOR
    PRINT NEW_LINE
END FUNCTION



FUNCTION shuff(REFERENCE ARRAY OF INT turn)
    // Initialize random number generator
    RANDOM_NUMBER_GENERATOR rng = INITIALIZE_WITH_RANDOM_DEVICE

    // Shuffle the elements of the 'turn' array
    SHUFFLE(turn, rng)

    // The 'nbPlayer' variable represents the number of players
    // Limit the shuffling to the number of players
    // This ensures that only the order of active players' turns is shuffled
    SHUFFLE(turn, FIRST 'nbPlayer' ELEMENTS OF 'turn' ARRAY, rng)
END FUNCTION
